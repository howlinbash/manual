
# install git to projects home directory
git init

# all steps to 1st commit. don't push to keep offline.
git status
git add -A
git commit -m "first commit"
git push

# make a branch
git branch first-edit
git checkout first-edit

# merge the branch. push/pull for online repository.
git checkout master	
git pull
git merge master
git push

# delete branch
git branch -D first-edit

# push a branch
git branch another-edit
git checkout another-edit
git add --all
git commit -m "another-edit"
git push origin another-edit 		   

#list all commits
git log

#revert to last commit -delete all changes since!
git reset --hard e5c89cf60e2d63c50f94bc628ce0ae8da4a31b4c

git show --summary

git diff --name-only hash1 hash2

# Felix's Merge
git checkout -b new-branch
git commit -m "Made some changes"
git checkout master
git pull  # just in case there are any changes pending
git merge new-branch
git branch -d new-branch  # delete the branch to tidy up

git stash
git stash list
git stash show
git stash pop
git stash drop

git reset --hard

# Playing with one file

# What's the difference between my local uncommited file and the last commit
git diff routes/index.js

# Delete those recent changes and revert to last commit
git checkout routes/index.js

git merge master pigeon-comments

# Say you wish to remove your last commit
git reset --soft HEAD~1
# Will do so without fucking your files
git reset
# will unstage those files that were commited
git checkout file
# will revert file back to old commit

# push local branch to hub
git push -u origin branch-name

# delete branch on hub
git push --delete origin pigeon-comments
